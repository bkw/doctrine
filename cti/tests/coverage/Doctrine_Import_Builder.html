690         }
691
692         $abstract = isset($definition['abstract']) && $definition['abstract'] === true ? 'abstract ':null;

697             $tableDefinitionCode = $this->buildTableDefinition($definition);

699         } else {
700             $tableDefinitionCode = null;
701             $setUpCode = null;
702         }
703         
704         $accessorsCode = (isset($definition['generate_accessors']) && $definition['generate_accessors'] === true) ? $this->buildAccessors($definition):null;
705         
706         $content = sprintf(self::$_tpl, $abstract,
707                                        $className,
708                                        $extends,
709                                        $tableDefinitionCode,
710                                        $setUpCode,
711                                        $accessorsCode);
712         
713         return $content;

716     /**
717      * buildRecord
718      *
719      * @param array $options 
720      * @param array $columns 
721      * @param array $relations 
722      * @param array $indexes 
723      * @param array $attributes 
724      * @param array $templates 
725      * @param array $actAs 
726      * @return void=
727      */
728     public function buildRecord(array $definition)
729     {
730         if ( ! isset($definition['className'])) {
731             throw new Doctrine_Import_Builder_Exception('Missing class name.');
732         }
733         
734         $definition['topLevelClassName'] = $definition['className'];
735         
736         if ($this->generateBaseClasses()) {
737             $definition['is_package'] = (isset($definition['package']) && $definition['package']) ? true:false;
738             
739             if ($definition['is_package']) {
740                 $e = explode('.', $definition['package']);
741                 $definition['package_name'] = $e[0];
742                 
743                 $definition['package_path'] = ! empty($e) ? implode(DIRECTORY_SEPARATOR, $e):$definition['package_name'];
744             }
745             
746             // Top level definition that extends from all the others
747             $topLevel = $definition;
748             unset($topLevel['tableName']);
749             
750             // If we have a package then we need to make this extend the package definition and not the base definition
751             // The package definition will then extends the base definition
752             $topLevel['inheritance']['extends'] = (isset($topLevel['package']) && $topLevel['package']) ? $this->_packagesPrefix . $topLevel['className']:'Base' . $topLevel['className'];
753             $topLevel['no_definition'] = true;
754             $topLevel['generate_once'] = true;
755             $topLevel['is_main_class'] = true;
756             unset($topLevel['connection']);
757
758             // Package level definition that extends from the base definition
759             if (isset($definition['package'])) {
760                 
761                 $packageLevel = $definition;
762                 $packageLevel['className'] = $topLevel['inheritance']['extends'];
763                 $packageLevel['inheritance']['extends'] = 'Base' . $topLevel['className'];
764                 $packageLevel['no_definition'] = true;
765                 $packageLevel['abstract'] = true;
766                 $packageLevel['override_parent'] = true;
767                 $packageLevel['generate_once'] = true;
768                 $packageLevel['is_package_class'] = true;
769                 unset($packageLevel['connection']);
770                 
771                 $packageLevel['tableClassName'] = $packageLevel['className'] . 'Table';
772                 $packageLevel['inheritance']['tableExtends'] = isset($definition['inheritance']['extends']) ? $definition['inheritance']['extends'] . 'Table':'Doctrine_Table';
773                 
774                 $topLevel['tableClassName'] = $topLevel['topLevelClassName'] . 'Table';
775                 $topLevel['inheritance']['tableExtends'] = $packageLevel['className'] . 'Table';
776             } else {
777                 $topLevel['tableClassName'] = $topLevel['className'] . 'Table';
778                 $topLevel['inheritance']['tableExtends'] = isset($definition['inheritance']['extends']) ? $definition['inheritance']['extends'] . 'Table':'Doctrine_Table';
779             }
780
781             $baseClass = $definition;
782             $baseClass['className'] = 'Base' . $baseClass['className'];
783             $baseClass['abstract'] = true;
784             $baseClass['override_parent'] = false;
785             $baseClass['is_base_class'] = true;
786
787             $this->writeDefinition($baseClass);
788             
789             if ( ! empty($packageLevel)) {
790                 $this->writeDefinition($packageLevel);
791             }
792             
793             $this->writeDefinition($topLevel);
794         } else {
795             $this->writeDefinition($definition);
796         }
797     }

826     /**
827      * writeDefinition
828      *
829      * @param array $options 
830      * @param array $columns 
831      * @param array $relations 
832      * @param array $indexes 
833      * @param array $attributes 
834      * @param array $templates 
835      * @param array $actAs 
836      * @return void
837      */
838     public function writeDefinition(array $definition)
839     {
840         $definitionCode = $this->buildDefinition($definition);
841
842         $fileName = $definition['className'] . $this->_suffix;
843
844         $packagesPath = $this->_packagesPath ? $this->_packagesPath:$this->_path;
845
846         // If this is a main class that either extends from Base or Package class
847         if (isset($definition['is_main_class']) && $definition['is_main_class']) {
848             // If is package then we need to put it in a package subfolder
849             if (isset($definition['is_package']) && $definition['is_package']) {
850                 $writePath = $this->_path . DIRECTORY_SEPARATOR . $definition['package_name'];
851             // Otherwise lets just put it in the root of the path
852             } else {
853                 $writePath = $this->_path;
854             }
855
856             $this->writeTableDefinition($definition['tableClassName'], $writePath, array('extends' => $definition['inheritance']['tableExtends']));
857         }
858         // If is the package class then we need to make the path to the complete package
859         else if (isset($definition['is_package_class']) && $definition['is_package_class']) {
860             $writePath = $packagesPath . DIRECTORY_SEPARATOR . $definition['package_path'];
861
862             $this->writeTableDefinition($definition['tableClassName'], $writePath, array('extends' => $definition['inheritance']['tableExtends']));
863         }
864         // If it is the base class of the doctrine record definition
865         else if (isset($definition['is_base_class']) && $definition['is_base_class']) {
866             // If it is a part of a package then we need to put it in a package subfolder
867             if (isset($definition['is_package']) && $definition['is_package']) {
868                 $basePath = $this->_path . DIRECTORY_SEPARATOR . $definition['package_name'];
869                 $writePath = $basePath . DIRECTORY_SEPARATOR . $this->_baseClassesDirectory;
870             // Otherwise lets just put it in the root generated folder
871             } else {
872                 $writePath = $this->_path . DIRECTORY_SEPARATOR . $this->_baseClassesDirectory;
873             }
874         }
875
876         // If we have a writePath from the if else conditionals above then use it
877         if (isset($writePath)) {
878             Doctrine_Lib::makeDirectories($writePath);
879
880             $writePath .= DIRECTORY_SEPARATOR . $fileName;
881         // Otherwise none of the conditions were met and we aren't generating base classes
882         } else {
883             Doctrine_Lib::makeDirectories($this->_path);
884
885             $writePath = $this->_path . DIRECTORY_SEPARATOR . $fileName;
886         }
887
888         $code = "<?php" . PHP_EOL;
889
890         if (isset($definition['connection']) && $definition['connection']) {
891             $code .= "// Connection Component Binding\n";
892             $code .= "Doctrine_Manager::getInstance()->bindComponent('" . $definition['connectionClassName'] . "', '" . $definition['connection'] . "');\n";
893         }
894
895         $code .= PHP_EOL . $definitionCode;
896
897         if (isset($definition['generate_once']) && $definition['generate_once'] === true) {
898             if ( ! file_exists($writePath)) {
899                 $bytes = file_put_contents($writePath, $code);
900             }
901         } else {
902             $bytes = file_put_contents($writePath, $code);
903         }
904
905         if (isset($bytes) && $bytes === false) {
906             throw new Doctrine_Import_Builder_Exception("Couldn't write file " . $writePath);
907         }
908
909         Doctrine::loadModel($definition['className'], $writePath);
910     }
