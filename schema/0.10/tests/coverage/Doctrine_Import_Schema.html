496                     // Check if the column we are inflecting has a _id on the end of it before trying to inflect it and find
497                     // the class name for the column
498                     if (strpos($column['name'], '_id')) {
499                         $columnClassName = Doctrine_Inflector::classify(str_replace('_id', '', $column['name']));
500                         if (isset($array[$columnClassName]) && !isset($array[$className]['relations'][$columnClassName])) {
501                             $array[$className]['relations'][$columnClassName] = array();
502                         }
503                     }
504                 }
505             }
506         }
507
508         foreach ($array as $name => $properties) {
509             if ( ! isset($properties['relations'])) {
510                 continue;
511             }
512             
513             $className = $properties['className'];
514             $relations = $properties['relations'];
515             
516             foreach ($relations as $alias => $relation) {
517                 $class = isset($relation['class']) ? $relation['class']:$alias;
518                 
519                 $relation['class'] = $class;
520                 $relation['alias'] = isset($relation['alias']) ? $relation['alias'] : $alias;
521                 
522                 // Attempt to guess the local and foreign
523                 if (isset($relation['refClass'])) {
524                     $relation['local'] = isset($relation['local']) ? $relation['local']:Doctrine::tableize($name) . '_id';
525                     $relation['foreign'] = isset($relation['foreign']) ? $relation['foreign']:Doctrine::tableize($class) . '_id';
526                 } else {
527                     $relation['local'] = isset($relation['local']) ? $relation['local']:Doctrine::tableize($relation['class']) . '_id';
528                     $relation['foreign'] = isset($relation['foreign']) ? $relation['foreign']:'id';
529                 }
530                 
531                 if (isset($relation['refClass'])) {
532                     $relation['type'] = 'many';
533                 }
534                 
535                 if (isset($relation['type']) && $relation['type']) {
536                     $relation['type'] = $relation['type'] === 'one' ? Doctrine_Relation::ONE:Doctrine_Relation::MANY;
537                 } else {
538                     $relation['type'] = Doctrine_Relation::ONE;
539                 }
540
541                 if (isset($relation['foreignType']) && $relation['foreignType']) {
542                     $relation['foreignType'] = $relation['foreignType'] === 'one' ? Doctrine_Relation::ONE:Doctrine_Relation::MANY;
543                 }
544                 
545                 $relation['key'] = $this->_buildUniqueRelationKey($relation);
546                 
547                 $this->_validateSchemaElement('relation', array_keys($relation));
548                 
549                 $this->_relations[$className][$alias] = $relation;
550             }
551         }
552         
553         // Now we auto-complete opposite ends of relationships
554         $this->_autoCompleteOppositeRelations();
555         
556         // Make sure we do not have any duplicate relations
557         $this->_fixDuplicateRelations();
558
559         // Set the full array of relationships for each class to the final array
560         foreach ($this->_relations as $className => $relations) {
561             $array[$className]['relations'] = $relations;
562         }
563         
564         return $array;
565     }
566
567     /**
568      * fixRelationships
569      *
570      * Loop through all relationships building the opposite ends of each relationship
571      * and make sure no duplicate relations exist
572      *
573      * @return void
574      */
575     protected function _autoCompleteOppositeRelations()
576     {
577         foreach($this->_relations as $className => $relations) {
578             foreach ($relations AS $alias => $relation) {
579                 if ((isset($relation['equal']) && $relation['equal']) || (isset($relation['autoComplete']) && $relation['autoComplete'] === false)) {
580                     continue;
581                 }
582                 
583                 $newRelation = array();
584                 $newRelation['foreign'] = $relation['local'];
585                 $newRelation['local'] = $relation['foreign'];
586                 $newRelation['class'] = isset($relation['foreignClass']) ? $relation['foreignClass']:$className;
587                 $newRelation['alias'] = isset($relation['foreignAlias']) ? $relation['foreignAlias']:$className;
588                 
589                 // this is so that we know that this relation was autogenerated and
590                 // that we do not need to include it if it is explicitly declared in the schema by the users.
591                 $newRelation['autogenerated'] = true; 
592                 
593                 if (isset($relation['refClass'])) {
594                     $newRelation['refClass'] = $relation['refClass'];
595                     $newRelation['type'] = isset($relation['foreignType']) ? $relation['foreignType']:$relation['type'];
596                 } else {                
597                     if(isset($relation['foreignType'])) {
598                         $newRelation['type'] = $relation['foreignType'];
599                     } else {
600                         $newRelation['type'] = $relation['type'] === Doctrine_Relation::ONE ? Doctrine_Relation::MANY:Doctrine_Relation::ONE;
601                     }
602                 }
603
604                 // Make sure it doesn't already exist
605                 if ( ! isset($this->_relations[$relation['class']][$newRelation['alias']])) {
606                     $newRelation['key'] = $this->_buildUniqueRelationKey($newRelation);
607                     $this->_relations[$relation['class']][$newRelation['alias']] = $newRelation;
608                 }
609             }
610         }
611     }
612
613     /**
614      * _fixDuplicateRelations
615      *
616      * Ensure the relations for each class are unique and that no duplicated relations exist from the auto generated relations
617      * and the user explicitely defining the opposite end
618      *
619      * @return void
620      */
621     protected function _fixDuplicateRelations()
622     {
623         foreach($this->_relations as $className => $relations) {
624             // This is for checking for duplicates between alias-relations and a auto-generated relations to ensure the result set of unique relations
625             $existingRelations = array();
626             $uniqueRelations = array();
627             foreach ($relations as $relation) {
628                 if ( ! in_array($relation['key'], $existingRelations)) {
629                     $existingRelations[] = $relation['key'];
630                     $uniqueRelations = array_merge($uniqueRelations, array($relation['alias'] => $relation));
631                 } else {
632                     // check to see if this relationship is not autogenerated, if it's not, then the user must have explicitly declared it
633                     if ( ! isset($relation['autogenerated']) || $relation['autogenerated'] != true) {
634                         $uniqueRelations = array_merge($uniqueRelations, array($relation['alias'] => $relation));
635                     }
636                 }
637             }
638             
639             $this->_relations[$className] = $uniqueRelations;
640         }
641     }
642
643     /**
644      * _buildUniqueRelationKey
645      *
646      * Build a unique key to identify a relationship by
647      * Md5 hash of all the relationship parameters
648      *
649      * @param string $relation 
650      * @return void
651      */
652     protected function _buildUniqueRelationKey($relation)
653     {
654         return md5($relation['local'].$relation['foreign'].$relation['class'].(isset($relation['refClass']) ? $relation['refClass']:null));
655     }

657     /**
658      * _validateSchemaElement
659      *
660      * @param string $name 
661      * @param string $value 
662      * @return void
663      */
664     protected function _validateSchemaElement($name, $element)
665     {
666         $element = (array) $element;
667
668         $validation = $this->_validation[$name];
669
670         // Validators are a part of the column validation
671         // This should be fixed, made cleaner
672         if ($name == 'column') {
673             $validators = $this->getValidators();
674             $validation = array_merge($validation, $validators);
675         }
676
677         $validation = array_flip($validation);
678         foreach ($element as $key => $value) {
679             if ( ! isset($validation[$value])) {
680                 throw new Doctrine_Import_Exception('Invalid schema element named "' . $value . '"');
681             }
682         }
683     }
